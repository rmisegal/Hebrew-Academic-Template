מסמך זה מדגים שימוש בשוליים מותאמים אישית באמצעות פקודת
\en{\textbackslash geometry\{\}} המובנית בתבנית מגרסה \en{\clsversion}.

\vspace{2mm}

תכונות עיקריות:
\begin{itemize}
\item שוליים מותאמים אישית -- \en{top=10mm, bottom=10mm, left=15mm, right=15mm}
\item תמיכה מלאה ב-\en{BiDi} -- עברית ואנגלית משולבים
\item טבלאות \en{RTL} עם \en{\textbackslash hebcellc\{\}} ו-\en{\textbackslash rtltabular}
\item מספרים ב-\en{LTR}: \num{12345}, אחוזים: \percent{18}, שנים: \hebyear{2026}
\end{itemize}

\textbf{תחביר:}

\begin{english}
\begin{verbatim}
% In preamble (after \documentclass):
\geometry{top=10mm, bottom=10mm, left=15mm, right=15mm}

% In document body (for mid-document change):
\newgeometry{top=8mm, bottom=6mm, left=12mm, right=12mm}

% Restore original margins:
\restoregeometry
\end{verbatim}
\end{english}
